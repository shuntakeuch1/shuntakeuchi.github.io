PHPでスネークケースの変換 2016/11/11

PHPでキャメルケースからスネークケースに変換

<?php
// ファイルポインタを開く
$file = fopen("tmp.txt", "r");

// 1行ずつ読み込みスネークケースにして返す。
while ($line = fgets($file)) {
  echo strtolower(preg_replace('/[A-Z]/','_\0',$line));
}
// ファイルポインタを閉じる
fclose($file);
